// @audit-clean: MetaFlow earnings tracker utility

import AuditLog from '../models/AuditLog.js';

export const trackEarnings = async ({ module, action, entityId, performedBy }) => {
  try {
    await AuditLog.create({
      entity: 'Earnings',
      entityId,
      action: `meta:${module}:${action}`,
      performedBy,
      timestamp: new Date().toISOString(),
      source: 'metaFlow',
    });

    console.log(`üí∞ [MetaFlow] Earnings tracked: ${module}:${action}`);
  } catch (err) {
    console.error('‚ùå [MetaFlow] Failed to track earnings:', err.message);
  }
};
